화살표 함수 =>

        function plus1(){
            return 10;
        }
        //화살표 함수
        plus2=()=>{
            return 20;
        }
        document.write(plus1());
        document.write(plus2());



        function minus1(a, b){
            return a-b;
        }
        //화살표 함수
        minus2=(a, b)=>{
            return a-b;
        }
        document.write(minus1(10, 5));
        document.write(minus2(20, 10));
