Object

객체 생성 방법 2가지,
프로퍼티 추가 

        /객체생성 1
        let dic1 = new Object();
        dic1.boy="소년";
        document.write(`dic2.boy = ${dic1.boy}`);

        //객체생성 2
        let dic2 = {
            boy:"땅콩",
            girl:"오징어",
            friend:300
        }
        document.write(`dic2.boy = ${dic2.boy}`);

        //프로퍼티 추가
        dic2.apple="사과";
        document.write(`dic2.apple = ${dic2.apple}`);

        //프로퍼티 삭제
        delete dic2.boy;
        document.write(`dic2.boy = ${dic2.boy}`);   //undefined
