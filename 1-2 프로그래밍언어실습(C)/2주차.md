```c
#include <stdio.h>

// 운영체계가 호출하는 함수, void로 매개변수 없음을 표시
int main(void){
    puts("3장 첫 c 프로그램!\n");

    printf("키워드: int void return 등\n");
    printf("식별자: main puts 등\n");

    return 0;
}
```

```c
#include <stdio.h>

int main(void){
    int data1=20, data2=13;
    int diff = data1-data2;
    int sum = data1+data2;

    printf("data1: %d, data2: %d\n", data1, data2);
    printf("차: %d, 합: %d\n", diff, sum);

    return 0;
}
```
